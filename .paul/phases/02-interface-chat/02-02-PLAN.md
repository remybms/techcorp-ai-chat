    ```ts
    const OLLAMA_BASE = 'http://localhost:11434'
    const MODEL_NAME = 'phi3-financial'

    export async function checkConnection(): Promise<boolean> {
      try {
        const res = await fetch(OLLAMA_BASE, { signal: AbortSignal.timeout(3000) })
        return res.ok
      } catch {
        return false
      }
    }

    export async function streamCompletion(
      prompt: string,
      onToken: (token: string) => void,
      signal?: AbortSignal
    ): Promise<void> {
      const res = await fetch(`${OLLAMA_BASE}/api/generate`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          model: MODEL_NAME,
          prompt,
          stream: true,
          options: { temperature: 0.3, num_predict: 512, top_p: 0.9, repeat_penalty: 1.1 }
        }),
        signal
      })

      if (!res.ok) throw new Error(`Ollama error: ${res.status}`)
      if (!res.body) throw new Error('No response body')

      const reader = res.body.getReader()
      const decoder = new TextDecoder()

      while (true) {
        const { done, value } = await reader.read()
        if (done) break
        const lines = decoder.decode(value).split('\n').filter(Boolean)
        for (const line of lines) {
          try {
            const json = JSON.parse(line)
            if (json.response) onToken(json.response)
            if (json.done) return
          } catch { /* ligne partielle, ignorer */ }
        }
      }
    }
    ```

    ```ts
    const handleSubmit = async (e: React.FormEvent) => {
      e.preventDefault()
      const prompt = input.trim()
      if (!prompt) return

      const userMsg: Message = { id: Date.now(), role: 'user', content: prompt }
      const assistantMsg: Message = { id: Date.now() + 1, role: 'assistant', content: '', isLoading: true }

      setMessages(prev => [...prev, userMsg, assistantMsg])
      setInput('')
      setIsLoading(true)

      try {
        let accumulated = ''
        await streamCompletion(prompt, (token) => {
          accumulated += token
          setMessages(prev =>
            prev.map(m => m.id === assistantMsg.id
              ? { ...m, content: accumulated, isLoading: false }
              : m
            )
          )
        })
      } catch (err) {
        setMessages(prev =>
          prev.map(m => m.id === assistantMsg.id
            ? { ...m, content: 'Erreur : serveur Ollama indisponible. Vérifiez que `ollama serve` est actif.', isLoading: false }
            : m
          )
        )
        setIsConnected(false)
      } finally {
        setIsLoading(false)
      }
    }
    ```
